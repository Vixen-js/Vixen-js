> **SystemTrayIcon** is a widget that provides a system tray icon for your application.

## Example

export const qSystemTrayIconExample = `
import { SystemTrayIcon } from '@vixen-js/core-react';
import { QIcon } from '@vixen-js/core';
import { resolve } from 'path';

const icon = new QIcon(
  resolve('./assets/icon.png')
);

export function SystemTrayIconExample() {
    return (
        <SystemTrayIcon icon={icon} visible={true} />
    );
}
`;
