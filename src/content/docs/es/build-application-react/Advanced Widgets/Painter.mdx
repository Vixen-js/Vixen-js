export const qPainterExample = `
import { Window, View } from '@vixen-js/core-react';
import { QPainter } from '@vixen-js/core';

export const App = () => {

  const paint = () => {
    const painter = new QPainter();
    painter.drawText(40, 40, "Hola Painter");
    painter.end();
  }

  return (
    <Window 
      size={{ width: 300, height: 300 }} 
      onPaint={paint} />
  )
}
`;
