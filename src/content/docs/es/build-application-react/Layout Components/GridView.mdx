> **GridView** es un layout que organiza widgets en una grilla.

## Ejemplo

export const gridLayoutExample = `
import { GridView, GridRow, GridColumn, Text } from '@vixen-js/core-react';

export function gridLayoutExample() {
    return (
        <GridView>
            <GridRow>
                <GridColumn><Text>Hello World</Text></GridColumn>
                <GridColumn><Text>Hello World</Text></GridColumn>
            </GridRow>
            <GridRow>
                <GridColumn ><Text>Hello World</Text></GridColumn>
                <GridColumn ><Text>Hello World</Text></GridColumn>
            </GridRow>
        </GridView>
    )
}
`;
