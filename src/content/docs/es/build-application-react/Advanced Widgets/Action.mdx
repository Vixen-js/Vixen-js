> **Action** es un widget que proporciona una acción que puedes utilizar en tus menus.

## Ejemplo

export const qActionExample = `
import { Action, Menu } from '@vixen-js/core-react';

export const Menu = () => {
    return (
        <Menu title="Menu principal">
            <Action text="Mi Acción" />
        </Menu>
    )
}
`;
