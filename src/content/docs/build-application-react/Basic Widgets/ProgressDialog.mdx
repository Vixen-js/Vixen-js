> **ProgressDialog** is a widget that provides a progress dialog that you can use in your application.

## Example

export const qProgressDialogExample = `
import { ProgressDialog } from '@vixen-js/core-react';
import { useState } from "react";

export function ProgressDialogExample() {
    const [isOpen, setIsOpen] = useState(true);
    return (
        <>
          <ProgressDialog 
            open={isOpen}
            maxValue={100}
            minValue={0}
            value={50} />
          <Button on={{clicked: () => setIsOpen(!isOpen)}} text="Open/Close" />
        </>
    );
}
`;
