> **Action** is a widget that provides an action that you can use in your menus.

## Example

export const qActionExample = `
import { Action, Menu } from '@vixen-js/core-react';

export const Menu = () => {
    return (
        <Menu title="Main Menu">
            <Action text="Your action" />
        </Menu>
    )
}
`;
