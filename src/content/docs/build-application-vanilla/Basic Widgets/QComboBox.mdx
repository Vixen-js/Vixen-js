> **QComboBox** is a widget that provides a selectable drop down menu

## Example

export const qComboBoxExample = `
import { QComboBox } from '@vixen-js/core';

const combobox = new QComboBox();
combobox.addItems(["Item 1", "Item 2", "Item 3"]);

combobox.addEventListener("currentTextChanged", (text) => {
    console.log(\`Text Changed to: \${text}\`);
});

comboboxaddEventListener("currentIndexChanged", (idx) => {
    console.log(\`Index Changed to: \${idx}\`);
});

`;
