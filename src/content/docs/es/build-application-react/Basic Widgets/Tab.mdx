> **Tab** es un widget que proporciona un componente Tab que puedes utilizar en tu aplicación.

## Ejemplo

export const tabExample = `
import { Tab, TabItem } from '@vixen-js/core-react';

export function TabExample() {
    return (
        <Tab>
            <TabItem text="Tab 1" />
            <TabItem text="Tab 2" />
            <TabItem text="Tab 3" />
        </Tab>
    );
}
`;
