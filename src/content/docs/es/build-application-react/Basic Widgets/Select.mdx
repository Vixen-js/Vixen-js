> **Select** es un widget que proporciona un selectable drop down menu.

## Ejemplo

export const selectExample = `
import { Select, SelectItem } from '@vixen-js/core-react';

export function SelectExample() {
    const Items: SelectItem[] = [
        { text: "Item 1" },
        { text: "Item 2" },
        { text: "Item 3" }
    ];
    return (
        <Select items={Items} />
    );
}
`;
